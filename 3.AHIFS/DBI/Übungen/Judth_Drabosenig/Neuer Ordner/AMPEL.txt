int counter = 0;
int oldcounter = 0;
int Delay = 20;

void setup() {
  pinMode(6,INPUT_PULLUP);
  Serial.begin(9600);
  attachInterrupt(digitalPinToInterrupt(6),myInterruptfunction,FALLING);
}

void loop() {

  if(millis() - interruptTime > Delay){
    attachInterrupt(digitalPinToInterrupt(6),myInterruptfunction,FALLING);
  }
  if(counter > oldcounter){
    Serial.print(counter);
    oldcounter = counter;
  } 
}

void myInterruptfunction(){
  detachInterrupt(digitalPinToInterrupt(6));
  interruptTime = millis();
  counter++;
}
