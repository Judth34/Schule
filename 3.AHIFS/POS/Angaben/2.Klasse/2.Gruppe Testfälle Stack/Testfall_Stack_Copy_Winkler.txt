private static bool testfall_Copy_Christian_Winkler(SimpleStack firstStack, SimpleStack secondStack, int groessestack, int minimum)
{
    int boolCounter = 0;
    bool result = false;
    fillStack(firstStack, groessestack, minimum);
    secondStack = firstStack.Copy();
    for(int StackIdx = 0; StackIdx < groessestack; StackIdx++)
    {
        if(firstStack.getStack()[StackIdx] == secondStack.getStack()[StackIdx])
            {
                boolCounter++;
            }
        }
        if(boolCounter == groessestack)
        {
            result = true;
        }
        return result;
    }
}
